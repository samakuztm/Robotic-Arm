# 3/5/2026 3:06 PM - Basic Code for the Arm  

_Time spent: 5.0h_  

Here is some basic code I think ill use for the robotic arm, I used some help from chatgpt for the parts I don't completely understand yet.

Basically here's how it works.

When you press a key on your keyboard, the Python program detects it and sends a single character to the ESP32.

Therefore I will be able to control the arm from my computer. Later on I can make it autonomous and stuff but for now i think this is a good starting point.

#include <ESP32Servo.h>

Servo baseServo;
Servo shoulderServo;
Servo elbowServo;
Servo gripperServo;

int basePos = 90;
int shoulderPos = 90;
int elbowPos = 90;
int gripperPos = 90;

void setup() {
  Serial.begin(115200);

  baseServo.attach(13);
  shoulderServo.attach(12);
  elbowServo.attach(14);
  gripperServo.attach(27);

  baseServo.write(basePos);
  shoulderServo.write(shoulderPos);
  elbowServo.write(elbowPos);
  gripperServo.write(gripperPos);
}

void loop() {

  if (Serial.available()) {
    char key = Serial.read();

    if (key == 'a') basePos += 5;
    if (key == 'd') basePos -= 5;

    if (key == 'w') shoulderPos += 5;
    if (key == 's') shoulderPos -= 5;

    if (key == 'i') elbowPos += 5;
    if (key == 'k') elbowPos -= 5;

    if (key == 'o') gripperPos += 5;
    if (key == 'l') gripperPos -= 5;

    basePos = constrain(basePos, 0, 180);
    shoulderPos = constrain(shoulderPos, 0, 180);
    elbowPos = constrain(elbowPos, 0, 180);
    gripperPos = constrain(gripperPos, 0, 180);

    baseServo.write(basePos);
    shoulderServo.write(shoulderPos);
    elbowServo.write(elbowPos);
    gripperServo.write(gripperPos);
  }

}
